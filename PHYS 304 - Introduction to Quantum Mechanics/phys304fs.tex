\section{The Wave Function}
Time dependent Schrodinger Equation:\\
\tab $\ds i\hbar \pdv{\Psi}{t} = -\frac{\hbar^2}{2m}\pdv[2]{\Psi}{x}+ V\Psi$

Standard Deviation:\\
\tab $\sigma = \sqrt{\expval{j^2} - \expval{j}^2}$

Momentum: \\
\tab $\ds \expval{p} = m \dv{\expval{x}}{t} = \int\Psi^* \left(\frac{\hbar}{i}\pdv{}{x}\right)\Psi\,dx$

\section{Infinite Square Well}
Time Independent Schrodinger Equation:\\
\tab $\ds -\frac{\hbar^2}{2m}\pdv[2]{\psi}{x}+ V\psi = E\psi$

Eigenstate Expansion:\\
\tab $\ds \Psi(x,t) = \sum_{n=1}^{\infty} c_n \psi_n(x)e^{-iE_nt/\hbar} = \sum_{n=1}^{\infty} c_n \Psi_n(x,t)$

Energy In Infinite Square Well:\\
\tab $\ds E_n = \frac{n^2\pi^2\hbar^2}{2ma^2}$

Stationary States:\\
\tab $\ds \psi_n(x) = \sqrt{\frac{2}{a}}\sin\left(\frac{n\pi}{a}x\right)$

Determining Coefficients:\\
\tab $\ds c_n = \sqrt{\frac{2}{a}}\int_{0}^{a}\sin\left(\frac{n\pi}{a}x\right)\Psi(x,0)\,dx$

Expectation Value of Energy:\\
\tab $\expval{H} = \sum_{n=1}^{\infty}\abs{c_n}^2E_n$

\section{Harmonic Oscillator}
\tab $k = \omega^2m$

Ladder Operators:\\
\tab $a_\pm = \frac{1}{\sqrt{2\hbar m\omega}}(\mp ip + m\omega x)$\\
\tab $\ap \psi_n = \sqrt{n+1}\,\psi_{n+1}$\\
\tab $\am \psi_n = \sqrt{n}\,\psi_{n-1}$

Operators:\\
\tab $\ds x = \sqrt{\frac{\hbar}{2m\omega}}(\ap + \am)$\\
\tab $\ds p = i\sqrt{\frac{\hbar m\omega}{2}}(\ap - \am)$\\
\tab $\ds x^2 = \frac{\hbar}{2m\omega}\left[(\ap)^2 + (\ap\am) + (\am\ap) + (\am)^2\right]$

Commutation:\\
\tab $\comm{x}{p} = i\hbar$\\
\tab $\comm{\am}{\ap} = 1$

Hamiltonian:\\
\tab $\ds H = \hbar \omega \left(\ap \am + \frac{1}{2}\right)$\\
\tab $\ds \ap \am + \am\ap = 2 \left(\frac{H}{\hbar \omega}\right)$\\

States:\\
\tab $\ds \psi_0(x) = \left(\frac{m\omega}{\pi\hbar}\right)^{1/4}e^{-\frac{m\omega}{2\hbar}x^2}$\\
\tab $\ds \psi_n = \frac{1}{\sqrt{n!}}(\ap)^n\psi_0$

Energy:\\
\tab $E_n = \left(n + \frac{1}{2}\right)\hbar\omega$

\section{Generalized Statistical Interpretation}
Momentum Expansion:\\
\tab $\ds \Phi(p,t) = \frac{1}{\sqrt{2\pi\hbar}}\int_{-\infty}^{\infty}e^{-ipx/\hbar}\,\Psi(x,t)\,dx$\\
\tab $\ds \Psi(x,t) = \frac{1}{\sqrt{2\pi\hbar}}\int_{-\infty}^{\infty}e^{ipx/\hbar}\,\Phi(p,t)\,dp$

\section{Trig Identities}
\tab $2\cos\theta\cos\phi = \cos(\theta - \phi) + \cos(\theta + \phi)$\\
\tab $2\sin\theta\sin\phi = \cos(\theta - \phi) - \cos(\theta + \phi)$\\
\tab $2\sin\theta\cos\phi = \sin(\theta + \phi) + \sin(\theta - \phi)$\\
\tab $2\cos\theta\sin\phi = \sin(\theta + \phi) - \sin(\theta - \phi)$\\
