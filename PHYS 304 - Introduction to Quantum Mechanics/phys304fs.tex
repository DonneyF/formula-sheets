\section{The Wave Function}
Time dependent Schrodinger Equation:\\
\tab $\ds i\hbar \pdv{\Psi}{t} = -\frac{\hbar^2}{2m}\pdv[2]{\Psi}{x}+ V\Psi$

Standard Deviation:\\
\tab $\sigma = \sqrt{\expval{j^2} - \expval{j}^2}$

Momentum: \\
\tab $\ds \expval{p} = m \dv{\expval{x}}{t} = \int\Psi^* \left(\frac{\hbar}{i}\pdv{}{x}\right)\Psi\,dx$

\section{Infinite Square Well}
Time Independent Schrodinger Equation:\\
\tab $\ds -\frac{\hbar^2}{2m}\pdv[2]{\psi}{x}+ V\psi = E\psi$

Eigenstate Expansion:\\
\tab $\ds \Psi(x,t) = \sum_{n=1}^{\infty} c_n \psi_n(x)e^{-iE_nt/\hbar} = \sum_{n=1}^{\infty} c_n \Psi_n(x,t)$

Energy In Infinite Square Well:\\
\tab $\ds E_n = \frac{n^2\pi^2\hbar^2}{2ma^2}$

Stationary States:\\
\tab $\ds \psi_n(x) = \sqrt{\frac{2}{a}}\sin\left(\frac{n\pi}{a}x\right)$

Determining Coefficients:\\
\tab $\ds c_n = \sqrt{\frac{2}{a}}\int_{0}^{a}\sin\left(\frac{n\pi}{a}x\right)\Psi(x,0)\,dx$

Expectation Value of Energy:\\
\tab $\expval{H} = \sum_{n=1}^{\infty}\abs{c_n}^2E_n$

\section{Harmonic Oscillator}
\tab $k = \omega^2m$

Ladder Operators:\\
\tab $a_\pm = \frac{1}{\sqrt{2\hbar m\omega}}(\mp ip + m\omega x)$\\
\tab $\ap \psi_n = \sqrt{n+1}\,\psi_{n+1}$\\
\tab $\am \psi_n = \sqrt{n}\,\psi_{n-1}$\\

Operators:\\
\tab $\ds x = \sqrt{\frac{\hbar}{2m\omega}}(\ap + \am)$\\
\tab $\ds p = i\sqrt{\frac{\hbar m\omega}{2}}(\ap - \am)$\\
\tab $\ds x^2 = \frac{\hbar}{2m\omega}\left[(\ap)^2 + (\ap\am) + (\am\ap) + (\am)^2\right]$

Commutation:\\
\tab $\comm{x}{p} = i\hbar$\\
\tab $\comm{\am}{\ap} = 1$

Hamiltonian:\\
\tab $\ds H = \hbar \omega \left(\ap \am + \frac{1}{2}\right)$\\
\tab $\ds \ap \am + \am\ap = 2 \left(\frac{H}{\hbar \omega}\right)$\\

States:\\
\tab $\ds \psi_0(x) = \left(\frac{m\omega}{\pi\hbar}\right)^{1/4}e^{-\frac{m\omega}{2\hbar}x^2}$\\
\tab $\ds \psi_n = \frac{1}{\sqrt{n!}}(\ap)^n\psi_0$

Energy:\\
\tab $E_n = \left(n + \frac{1}{2}\right)\hbar\omega$

\section{Generalized Statistical Interpretation}
Momentum Expansion:\\
\tab $\ds \Phi(p,t) = \frac{1}{\sqrt{2\pi\hbar}}\int_{-\infty}^{\infty}e^{-ipx/\hbar}\,\Psi(x,t)\,dx$\\
\tab $\ds \Psi(x,t) = \frac{1}{\sqrt{2\pi\hbar}}\int_{-\infty}^{\infty}e^{ipx/\hbar}\,\Phi(p,t)\,dp$

\section{Angular Momentum}
$L_x = yp_z -zp_y$ \quad $L_y = zp_x - xp_z$ \quad $L_z = xp_y - yp_x$

Commutators:\\
\tab $\comm{L_x}{L_y} = i\hbar L_z$ \quad $\comm{L_y}{L_z} = i\hbar L_x$ \quad $\comm{L_z}{L_x} = i\hbar L_y$\\
\tab $\comm{L_z}{x} = i\hbar y$ \quad $\comm{L_z}{y} = -i\hbar x$ \quad $\comm{L_z}{z} = 0$\\
\tab $\comm{L_z}{p_x} = i\hbar p_y$ \quad $\comm{L_z}{p_y} = -i\hbar p_x$ \quad $\comm{L_z}{p_z} = 0$

Square of Angular Momentum:\\
\tab $L^2 \equiv L_x^2 + L_y^2 + L_z^2$ \qquad $\comm{L^2}{\vb{L}} = 0$

Ladder Operators:\\
\tab $L_\pm = L_x \pm iL_y$ \qquad $\comm{L_z}{L_\pm} = \pm i\hbar L_\pm$\\
\tab $L^2 = L_\pm L_\mp + L_z^2 \mp \hbar L_z$\\
\tab $\ds L_x = \frac{L_+ + L_-}{2}$ \qquad $\ds L_y = \frac{L_+ - L_-}{2i}$\\
\tab $L_+ \ket{l,l} = 0$ \qquad $L_- \ket{l,-l} =0$

Eigenvalues:\\
\tab $L^2 \ket{l,m} = \hbar^2 l(l+1)\ket{l,m}$\\
\tab $L_z \ket{l,m} = \hbar m \ket{l,m}$\\
\tab $L_\pm \ket{l,m} = A_l^m\ket{l,m+1}$\\
\tab $A_l^m = \hbar \sqrt{l(l+1) - m(m\pm 1)}$

Expectation Value when $L_x\ket{\psi} = L_y\ket{\psi}$:\\
\tab $\ds \bra{\psi}L_x^2\ket{\psi} = \bra{\psi}L_y^2\ket{\psi} = \frac{1}{2}\bra{\psi}L^2 - L_z^2\ket{\psi}$

Spherical Coordinate Representation:\\
\tab $L_x = \ds \frac{\hbar}{i}\left(-\sin\phi \pdv{\theta} - \cos\phi\cot\theta\pdv{\phi}\right)$\\
\tab $L_y = \ds \frac{\hbar}{i}\left(+\cos\phi \pdv{\theta} - \sin\phi\cot\theta\pdv{\phi}\right)$\\
\tab $\ds L_z = \frac{\hbar}{i}\pdv{\theta}$\\
\tab $\ds L_\pm = \pm\hbar e^{\pm i \theta}\left(\pdv{\theta} \pm i\cot\theta\pdv{\phi}\right)$\\
\tab $\ds L_+L_- = -\hbar^2\left(\pdv[2]{\theta}+\cot\theta\pdv{\theta} + \cot^2\theta\pdv[2]{\phi} + i\pdv{\phi}\right)$\\
\tab $\ds L^2 = -\hbar^2\left[\frac{1}{\sin\theta}\pdv{\theta}\left(\sin\theta\pdv{\theta}\right) + \frac{1}{\sin^2\theta}\pdv[2]{\phi}\right]$


\section{Trig Identities}
\tab $2\cos\theta\cos\phi = \cos(\theta - \phi) + \cos(\theta + \phi)$\\
\tab $2\sin\theta\sin\phi = \cos(\theta - \phi) - \cos(\theta + \phi)$\\
\tab $2\sin\theta\cos\phi = \sin(\theta + \phi) + \sin(\theta - \phi)$\\
\tab $2\cos\theta\sin\phi = \sin(\theta + \phi) - \sin(\theta - \phi)$\\

\section*{Integral Identities}
\tab $\ds \int \sin ^{2}{ax}\,dx={\frac {x}{2}}-{\frac {1}{4a}}\sin 2ax+C$\\
\tab $\ds \int \sin ^{3}{ax}\,dx={\frac {\cos 3ax}{12a}}-{\frac {3\cos ax}{4a}}+C$\\
\tab $\ds \int \cos ^{2}{ax}\,dx={\frac {x}{2}}+{\frac {1}{4a}}\sin 2ax+C$\\
\tab $\ds \int \cos^3{ax}\,dx = \frac{\sin ax}{a} - \frac{\sin^3{ax}}{3a} + C$\\
\tab $\ds \int \cos^{n}(ax)\sin(ax)\,dx = -\frac{\cos^{n+1}(ax)}{a(n+1)} + C$\\
\tab $\ds \int \sin^{n}(ax)\cos(ax)\,dx= \frac{\sin^{n+1}(ax)}{a(n+1)} + C$\\
\tab $\ds \int_0^\pi \cos^{2n+1}(ax)\sin(ax)\,dx = 0 \qquad n = 0, 1, 2 \ldots$\\
\tab $\ds \int_0^\pi \cos^{2n}(x)\sin(x)\,dx = \frac{2}{2n+1} \qquad n = 0, 1, 2 \ldots$
