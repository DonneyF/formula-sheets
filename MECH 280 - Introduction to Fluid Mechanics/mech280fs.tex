\section{Vectors \& Matrices}
Inner Product (aka scalar or dot product):\\
\tab $\vb{a}\cdot\vb{b} = \abs{\vb{a}}\abs{\vb{b}}\cos\theta$

Cross Product (aka vector product):\\
\tab $\vb{a}\cross\vb{b} = \abs{\vb{a}}\abs{\vb{b}}\sin\theta\,\vu{n}$

Triple Product:\\
\tab $\vb{a} \times (\vb{b} \times \vb{c}) = (\vb{a} \cdot \vb{c})\vb{b} - (\vb{a} \cdot \vb{b})\vb{c}$\\
\tab $(\vb{a} \times \vb{b}) \cdot \vb{c} = (\vb{c} \times \vb{a}) \cdot \vb{b} = (\vb{b} \times \vb{c}) \cdot \vb{a}$

Outer Product (aka tensor product):\\
\tab $\vb{v} \otimes \vb{w} \begin{bmatrix}v_{1}w_{1}&&v_{1}w_{2}&&\cdots&&v_{1}w_{m}\\
v_{2}w_{1}&&v_{2}w_{2}&&\cdots &&v_{2}w_{m}\\
\vdots &&\vdots &&\ddots &&\vdots \\
v_{n}w_{1}&&v_{n}w_{2}&&\cdots &&v_{n}w_{m}\end{bmatrix}$

\section{Vector Derivatives}
\subsection{Cartesian}
Gradient:\\
\tab $\ds \grad{f} =  \pdv{f}{x}\vu{x} + \pdv{f}{y}\vu{y} + \pdv{f}{z}\vu{z}$

Divergence:\\
\tab $\ds \div{\vb{v}} = \pdv{v_x}{x} + \pdv{v_y}{y} + \pdv{v_z}{z}$

Curl:\\
\vspace{-3mm}
\tab $\ds \curl{\vb{v}} = \left(\pdv{v_z}{y} - \pdv{v_y}{z}\right)\vu{x} + \left(\pdv{v_x}{z} - \pdv{v_z}{x}\right)\vu{y} + \left(\pdv{v_y}{x} - \pdv{v_x}{y}\right)\vu{z}$

\subsection{Spherical}
($\theta$ is the polar angle.)

Gradient:\\
\tab $\ds \grad{f} = \pdv{f}{r}\vu{r} + \frac{1}{r}\pdv{f}{\theta}\vu*{\theta} + \frac{1}{r\sin\theta}\pdv{f}{\phi}\vu*{\phi}$

Divergence:\\
\tab $\ds \div{\vb{v}} = \frac{1}{r^2}\pdv{r}(r^2v_r) + \frac{1}{r\sin\theta}\pdv{\theta}(\sin\theta v_\theta) + \frac{1}{r\sin\theta}\pdv{v_\theta}{\phi}$

Curl:\\
\tab $\ds \curl{\vb{v}} = \frac{1}{r\sin\theta}\left[\pdv{\theta}(\sin\theta\, v_\phi)- \pdv{v_\theta}{\phi}\right]\vu{r}\,+$\\
\tab \tab $\ds \frac{1}{r}\left[\frac{1}{\sin\theta}\pdv{v_r}{\phi}-\pdv{r}(r v_\phi)\right]\vu*{\theta} + \frac{1}{r}\left[\pdv{r}(r v_\theta)-\pdv{v_r}{\theta}\right]\vu*{\phi}$

Tensor:\\
\tab $\grad\vb{u} = \begin{bmatrix}
\vspace{2mm}
\ds \pdv{u_x}{x} &&\ds \pdv{u_y}{x}&&\ds \pdv{u_z}{x}\\
\vspace{2mm}
\ds \pdv{u_x}{y} &&\ds \pdv{u_y}{y}&&\ds \pdv{u_z}{y} \\
\ds \pdv{u_x}{z} &&\ds \pdv{u_y}{z}&&\ds \pdv{u_z}{z}\\
\end{bmatrix}$

\subsection{Cylindrical}
Gradient:\\
\tab $\ds \grad{f} = \pdv{f}{s}\vu{s}+\frac{1}{s}\pdv{f}{\phi}\vu*{\phi}+\pdv{f}{z}\vu{z}$

Divergence:\\
\tab $\ds \div{\vb{v}} = \frac{1}{s}\pdv{s}(sv_s)+\frac{1}{s}\pdv{v_\phi}{\phi} + \pdv{v_z}{z}$

Curl:\\
\tab $\ds \curl{\vb{v}} = \left[\frac{1}{s}\pdv{v_z}{\phi}-\pdv{v_\phi}{z}\right]\vu{s}+\left[\pdv{v_s}{z}-\pdv{v_z}{s}\right]\vu*{\phi}+\frac{1}{s}\left[\pdv{s}(sv_\phi)-\pdv{v_s}{\phi}\right]\vu{z}$

\section{Vector Calculus}
Divergence Theorem:\\
\tab $\ds \int (\div \vb{A})\, dV = \oint \vb{A}\cdot d\vb{S}$

Stoke's Theorem:\\
\tab $\ds \int (\curl{\vb{A}})\cdot d\vb{S} = \oint \vb{A}\cdot d\vb{l}$

Directional Derivative (slope of a scalar field in the direction of the vector $\vb{u}$):\\
\tab $\grad f \cdot \vb{u}$

Material Derivative:\\
\tab $\ds \mdv{\rho}{t} = \pdv{\rho}{t} + \vb{u}\cdot \grad\rho$
\tab $\ds \mdv{\vb{A}}{t} = \pdv{\vb{A}}{t} + \vb{u}\cdot \grad\vb{A}$

Liebniz Rule:\\
\tab $\ds \frac{d}{dt} \int_\text{CV(t)}\rho dV = \int_\text{CV}\pdv{\rho}{t}dt + \rho\int_\text{CS} \vb{u}_\text{CS}\cdot d\vb{S}$

\section{Integral Equations of Motion}
Flux:\\
\tab $\ds \phi = \int_\text{CS} \rho \vb{u}\cdot d\vb{S}$

Conservation of Mass:\\
\tab $\ds \int_\text{CV}\pdv{\rho}{t} = - \int_\text{CS} \rho \vb{u}\cdot d\vb{S}$

Conservation of Momentum:\\
$\ds \frac{d}{dt} \int_\text{CV}\rho \vb{u}dV = - \int_\text{CS}(\rho\vb{u})\vu{u}\cdot d\vb{S} + \int_\text{CV}\vb{f}_b\,dV + \int_\text{CS}\vb{t}\,dS$

\section{Differential Equations of Motion}
Conservation of Mass:\\
\tab $\ds \mdv{\rho}{t} + \rho\div{\vb{u}} = 0$

Conservation of Momentum:\\
\tab $\ds \rho \mdv{\vb{u}}{t} = \vb{f}_b + \vb{f}_c$
